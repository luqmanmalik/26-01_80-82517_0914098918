$ \begin{equation} \begin{split} \text{(ii)} \ \ \ \text{Proof:} \ \ (x-y)(x+y) &= [x+(-y)](x+y) = x(x+y)+(-y)(x+y) \\ &= x(x+y)-[y(x+y)] = x^2+xy-[yx+y^2] \\ &= x^2+[xy-yx]-y^2 \\ &= x^2-y^2. \\ & \ \square \end{split} \end{equation} $
$ \begin{equation} \begin{split} \text{(iii)} \ \ \text{Proof:} \ \ [x^2=y^2] \Rightarrow 0 &= y^2-x^2 \\ &= -x^2+y^2 = -(x^2-y^2) \\ &= x^2-y^2 \\ &= (x-y)(x+y) \\ &= x-y \ \ \ | \ \ \ \text{or} \ \ 0=x+y \Leftrightarrow x=-y. \\ x &= y. \\ & \ \square \end{split} \end{equation} $
$ \begin{equation} \begin{split} \text{(iv)} \ \ \text{Proof:} \ \ (x-y)(x^2+xy+y^2) &= x(x^2+xy+y^2)-y(x^2+xy+y^2) \\ &= x(x^2+xy+y^2)-[y(x^2+xy+y^2)] \\ &= x^3+x^2y+xy^2-[yx^2+xy^2+y^3] \\ &= x^3-y^3. \\ & \ \square \end{split} \end{equation} $
$ \begin{equation} \begin{split} \text{(v)} \ \ \text{Proof:} \ \ (x-y)(x^{n-1}+x^{n-2}y &+\cdots+xy^{n-2}+y^{n-1}) \\ &= x(x^{n-1}+x^{n-2}y +\cdots+xy^{n-2}+y^{n-1}) \\ & \ \ \ \ \ \ -[y(x^{n-1}+x^{n-2}y +\cdots+xy^{n-2}+y^{n-1})] \\ &= x^n+x^{n-1}y+\cdots+x^2y^{n-2}+xy^{n-1} \\ & \ \ \ \ \ \ -[x^{n-1}y+x^{n-2}y^2 +\cdots+xy^{n-1}+y^n] \\ &= x^n-y^n. \\ & \ \square \end{split} \end{equation} $

Using sigma notation, this proof may also be written as follows:

$ \begin{equation} \begin{split} (x-y) \cdot \sum_{j=0}^{n-1} x^jy^{n-1-j} &= x \bigg( \sum_{j=0}^{n-1} x^jy^{n-1-j} \bigg) - \bigg[ y \bigg( \sum_{j=0}^{n-1} x^jy^{n-1-j} \bigg) \bigg] \ \\ &= x^n + \sum_{j=0}^{n-2} x^{j+1}y^{n-1-j} - \bigg[ \sum_{j=1}^{n-1} x^jy^{n-j} + y^n \bigg] \ \\ &= x^n + \sum_{j=0}^{n-2} x^{j+1}y^{n-1-j} - \bigg[ \sum_{k=0}^{n-2} x^{k+1}y^{n-(k+1)} + y^n \bigg] \ \\ & \ \ \ \ \ \ \ \ \ \ \ (\text{where} \ k = j-1) \ \\ &= x^n-y^n. \\ & \ \square \end{split} \end{equation} $

$ \begin{array}{ll} \text{(i)} & \text{Proof:} \ \ \dfrac{ac}{bc} = \dfrac{a}{b} \cdot \dfrac{c}{c} = \dfrac{a}{b} \cdot 1 = \dfrac{a}{b}. \ \square \\ & \text{Proof:} \ \ a/b = ab^{-1} = (ac)(b^{-1}c^{-1}) = (ac)(bc)^{-1} \ (\text{by (iii)}) = ac/bc. \ \square \end{array} $
$ \begin{equation} \begin{split} \text{(ii)} \ \ \ \text{Proof:} \ \ \frac{a}{b} + \frac{c}{d} &= \bigg( \frac{d}{d} \bigg) \cdot \frac{a}{b} + \bigg( \frac{b}{b} \bigg) \cdot \frac{c}{d} \\ &= \frac{ad+bc}{bd}. \\ & \ \square \end{split} \end{equation} $
$ \begin{array}{ll} \text{(iii)} & \text{Proof:} \ \ (ab)^{-1} = 1/ab = (1/a) \cdot (1/b) = a^{-1}b^{-1}. \ \square \\ & \text{Proof:} \ \ ab(a^{-1}b^{-1}) = (a \cdot a^{-1})(b \cdot b^{-1}) = 1, \text{so} \ a^{-1} \cdot b^{-1} = (ab)^{-1}. \ \square \end{array} $
$ \begin{equation} \begin{split} \text{(iv)} \ \ \ \ \text{Proof:} \ \ (a/b)(c/d) &= (ab^{-1})(cd^{-1}) = (ac)(b^{-1}d^{-1}) \\ &= (ac)(bd)^{-1} (\text{by (iii)}) \\ &= (ac)/(bd). \\ & \ \square \end{split} \end{equation} $
$\text{(v)} \ \ \ \ \text{Proof:} \ \ \dfrac{a}{b} \bigg/ \dfrac{c}{d} = \dfrac{a}{b} \cdot \dfrac{d}{c} = \dfrac{ad}{bc}. \ \ \square$ $ \begin{equation} \begin{split} \ \ \ \ \ \ \ \ \text{Proof:} \ (a/b)/(c/d) &= (a/b)(c/d)^{-1} = (ab^{-1})(cd^{-1})^{-1} \\ &= (ab^{-1})(c^{-1}d) = ad(b^{-1}c^{-1}) \\ &= ad(bc)^{-1} = (ad)/(bc). \\ & \ \square \end{split} \end{equation} $
